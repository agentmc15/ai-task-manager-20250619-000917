```
def _ensure_archer_json_column(eng):
    """Add archer_json column to project table if it doesn't exist.

    Populated by the inline Archer mapping step at submit time, right
    after enrichment. Holds the 28-entry payload (22 enriched entries
    plus 6 *_archer_id / *_id entries) consumed by the downstream
    Archer publisher service.
    """
    with eng.connect() as conn:
        result = conn.execute(text("""
            SELECT column_name FROM information_schema.columns
            WHERE table_name = 'project' AND column_name = 'archer_json'
        """))
        if not result.fetchone():
            log.info("Adding archer_json column to project table...")
            conn.execute(text(
                "ALTER TABLE project ADD COLUMN archer_json JSONB NULL"
            ))
            conn.commit()
            log.info("archer_json column added")


# ---------------------------------------------------------------------------
# Seed data
```

**REPLACE WITH:**

```
    # Enrichment - runs inline, rolls back the whole submit on failure.
    # Archer mapping is part of the same transaction: if either step
    # fails, the whole submit is rolled back.
    try:
        enriched = enrich_project(project)
        project.enriched_json = enriched
        project.enriched = True
        # Tell SQLAlchemy the JSONB column changed (in-place list mutation
        # on a JSON column is not auto-detected by the ORM).
        flag_modified(project, "enriched_json")

        archer = build_archer_json(enriched)
        project.archer_json = archer
        flag_modified(project, "archer_json")
    except HTTPException:
        # A deliberate HTTPException from downstream should pass through.
        session.rollback()
        raise
```

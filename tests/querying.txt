    for row in raw_rows:
        if isinstance(row, list):
            # Row is a list of {col_id, value} dicts
            entry = {}
            for cell in row:
                if isinstance(cell, dict) and "col_id" in cell:
                    col_id = cell.get("col_id")
                    value = cell.get("value", "")
                    if col_id:
                        entry[col_id] = value
            # Skip completely empty rows

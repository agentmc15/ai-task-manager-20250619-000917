def _ensure_enriched_columns(eng):
    """Add enriched and enriched_json columns to project table if they don't exist.
    These are populated by the inline enrichment step at submit time. The
    enriched_json column holds the full 21-question payload consumed by the
    downstream Archer publisher service.
    """
    with eng.connect() as conn:
        # enriched
        result = conn.execute(text("""
            SELECT column_name FROM information_schema.columns
            WHERE table_name = 'project' AND column_name = 'enriched'
        """))
        if not result.fetchone():
            log.info("Adding enriched column to project table...")
            conn.execute(text(
                "ALTER TABLE project ADD COLUMN enriched BOOLEAN NOT NULL DEFAULT false"
            ))
            conn.commit()
            log.info("enriched column added (default false for existing rows)")

        # enriched_json
        result = conn.execute(text("""
            SELECT column_name FROM information_schema.columns
            WHERE table_name = 'project' AND column_name = 'enriched_json'
        """))
        if not result.fetchone():
            log.info("Adding enriched_json column to project table...")
            conn.execute(text(
                "ALTER TABLE project ADD COLUMN enriched_json JSONB NULL"
            ))
            conn.commit()
            log.info("enriched_json column added")

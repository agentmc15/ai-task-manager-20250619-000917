```python
# ---------------------------------------------------------
# SUBMISSION LIFECYCLE - user-facing and service-facing endpoints
# ---------------------------------------------------------


@project_router.post("/{project_id}/submit")
async def submit_project(
    project_id: str,
    user: CurrentUser = Depends(get_current_user),
    session: Session = Depends(get_session),
):
    """
    Transition a project from 'draft' to 'submitted'.

    Called by the frontend when the user clicks Submit for Review.
    After this transition, the project is locked - no further answers
    can be saved until an admin resets the status.
    """
    project = session.exec(
        select(Project).where(
            Project.id == project_id,
            Project.owner_email == user.email,
        )
    ).first()

    if not project:
        raise HTTPException(status_code=404, detail="Project not found")

    if project.status == "submitted":
        raise HTTPException(
            status_code=409,
            detail="Project has already been submitted.",
        )

    if project.status == "exported":
        raise HTTPException(
            status_code=409,
            detail="Project has already been exported and cannot be re-submitted.",
        )

    if project.status != "draft":
        raise HTTPException(
            status_code=409,
            detail=f"Project cannot be submitted from status '{project.status}'.",
        )

    project.status = "submitted"
    project.submitted_at = datetime.now(timezone.utc)
    session.add(project)
    session.commit()
    session.refresh(project)

    return {
        "project_id": str(project.id),
        "status": project.status,
        "submitted_at": project.submitted_at.isoformat() if project.submitted_at else None,
    }


@project_router.get("/ready-for-export")
async def list_submitted_projects(
    session: Session = Depends(get_session),
):
    """
    Return all projects currently in 'submitted' status, ordered by
    submitted_at ascending (oldest first).

    Called by the downstream enrichment/intermediary service to pick up
    work. No authentication required - this endpoint is intended for
    internal service-to-service traffic on the corporate network.

    Does NOT change any state. Projects stay in 'submitted' until the
    consuming service calls /project/{id}/mark-exported.
    """
    projects = session.exec(
        select(Project)
        .where(Project.status == "submitted")
        .order_by(Project.submitted_at)
    ).all()

    return [
        {
            "project_id": str(p.id),
            "title": p.title,
            "owner_email": p.owner_email,
            "submitted_at": p.submitted_at.isoformat() if p.submitted_at else None,
            "responses_json": p.responses_json or [],
        }
        for p in projects
    ]


@project_router.post("/{project_id}/mark-exported")
async def mark_project_exported(
    project_id: str,
    session: Session = Depends(get_session),
):
    """
    Transition a project from 'submitted' to 'exported'.

    Called by the downstream enrichment/intermediary service after it
    has successfully processed the project and pushed to Archer.
    No authentication required.

    Idempotent: calling this on an already-exported project returns 200 OK
    without changing state. Calling it on a 'draft' project returns 409.
    """
    project = session.get(Project, project_id)
    if not project:
        raise HTTPException(status_code=404, detail="Project not found")

    # Idempotent success for re-calls
    if project.status == "exported":
        return {
            "project_id": str(project.id),
            "status": "exported",
        }

    if project.status == "draft":
        raise HTTPException(
            status_code=409,
            detail="Cannot mark a draft project as exported. Project must be submitted first.",
        )

    project.status = "exported"
    session.add(project)
    session.commit()
    session.refresh(project)

    return {
        "project_id": str(project.id),
        "status": project.status,
    }
```

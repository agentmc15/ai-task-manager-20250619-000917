async function updateProject(projectId: string, data: {
    title?: string
    description?: string
  }) {
    const user = await getCurrentUser()
    const userId = user?.id || user?.email || user?.user_id
    if (!userId) {
      throw new Error('Cannot update project: no authenticated user')
    }
    return apiFetch<any>('/project/', {
      method: 'PUT',
      body: JSON.stringify({
        id: projectId,
        userId,
        ...data,
      }),
    })
  }
